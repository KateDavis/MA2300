\section{Measures of Center}
Mean
\begin{equation*}
\bar{x}=\frac{\sum\limits_{i=1}^{N} x_i }{N} 
\end{equation*}
\section{Measures of Spread}
Deviation
\begin{equation*}
Dev_{\bar{x}}=(x_i-\bar{x}) 
\end{equation*}
\begin{equation*}
Dev_{\bar{x}}^2=(x_i-\bar{x})^2 
\end{equation*}
Variance
\begin{multline*}
Var(X)=\frac{\sum_{i=1}^{N} Dev_{\bar{x}}^2}{N}=\frac{\sum_{i=1}^{N} (x_i-\bar{x})^2}{N}
\end{multline*}
Standard Deviation
\begin{equation*}
StdDev(X)=\sqrt{Var(X)} 
\end{equation*}
\section{Measures of Association}
Covariance
\begin{equation*}
Cov(X,Y)=\frac{\Sigma_{i=1}^{N} Dev_{\bar{x}}Dev_{\bar{y}}}{N}
\end{equation*}
Correlation
\begin{equation*}
Corr(X,Y)=\frac{Cov(Y,X)}{(StdDev(X)StdDev(Y))}
\end{equation*}
Regression Equation
\begin{equation*}
Y=\alpha + \beta X + \epsilon
\end{equation*}
\begin{equation*}
{\beta}=\frac{Cov(X,Y)}{Var(X)}=Corr(X,Y)\frac{StdDev(Y)}{StdDev(X)}
\end{equation*}
\begin{equation*}
{\alpha}=\bar{y}-{\beta}\bar{x}
\end{equation*}
